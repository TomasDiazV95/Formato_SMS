# Carga Archivo
archivo = files.upload()
nombre_archivo = list(archivo.keys())[0]

# --- LECTURA ROBUSTA: detectar la fila de encabezados real ("N° Contrato") ---
# Leemos sin encabezado para encontrar la fila que contiene los nombres de columnas
raw = pd.read_excel(nombre_archivo, header=None, engine="openpyxl")

# Buscar la fila donde aparece "N° Contrato" en cualquier columna
mask_header = raw.apply(lambda row: row.astype(str).str.strip().eq("N° Contrato").any(), axis=1)
if not mask_header.any():
    raise ValueError("No se encontró la fila de encabezados (que contenga 'N° Contrato'). Revisa el archivo.")

fila_header = mask_header.idxmax()  # índice de la primera coincidencia

# Usar esa fila como encabezado y tomar desde la fila siguiente como datos
headers = raw.iloc[fila_header].astype(str).str.strip().tolist()
df = raw.iloc[fila_header + 1:].copy()
df.columns = headers

# Eliminar filas completamente vacías y reindexar
df = df.dropna(how="all").reset_index(drop=True)

# (Opcional) normalizar algunos campos de fecha si existen
for col in ["Fecha Reporte", "Fecha de activación", "Fecha último pago"]:
    if col in df.columns:
        # el archivo suele venir con formato mm/dd/yyyy; si no aplica, queda NaT sin romper el flujo
        df[col] = pd.to_datetime(df[col], errors="coerce", dayfirst=False)

# Limpiar espacios en campos clave (si existen)
cols_strip = [
    "N° Contrato","Rut Cliente","Nombre Cliente","Tipo de Producto",
    "Marca Vehículo","Modelo Vehículo","DealerName","E-mail",
    "Región cliente","Ciudad","Comuna","Tramo de mora"
]
for c in cols_strip:
    if c in df.columns:
        df[c] = df[c].astype(str).str.strip()

# Asegurar la columna fija requerida para el CRM
df["NombreProducto"] = "AUTOMOTRIZ"
